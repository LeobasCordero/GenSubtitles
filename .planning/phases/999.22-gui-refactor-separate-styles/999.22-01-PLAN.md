```python
"""gensubtitles.gui.styles
~~~~~~~~~~~~~~~~~~~~~~~~~
Spacing constants, dimension constants, and widget-style helper functions
for the GenSubtitles desktop UI.

Companion to ``theme.py`` (colours + typography).  ``styles.py`` owns
layout values and widget-style application helpers so component code
stays focused on structure and behaviour.

Public API
----------
Constants
  SPACING_XS … SPACING_XXL  — integer pixel values for padx/pady
  BTN_HEIGHT_PRIMARY         — 44 px  (Generate, Translate, Save, Back, Clear)
  BTN_HEIGHT_CANCEL          — 36 px  (Cancel)
  BTN_HEIGHT_MINI            — 28 px  (colour-swatch buttons)
  BTN_WIDTH_BROWSE           — 80 px  (Browse…, Save as…)
  BTN_WIDTH_NARROW           — 100 px (Open Folder, other small buttons)
  BTN_WIDTH_SWATCH           — 36 px  (colour swatch square buttons)
  PROGRESS_BAR_HEIGHT        — 16 px

Helpers (apply-after-construction)
  apply_entry_style(widget)
  apply_accent_btn_style(widget)
  apply_secondary_btn_style(widget)
  apply_cancel_btn_style(widget)
  apply_progress_bar_style(widget)
  apply_stage_label_style(widget)
  apply_secondary_label_style(widget)
  apply_settings_header_style(widget)
  apply_window_bg(window)
"""
from __future__ import annotations

from .theme import font, p

# ---------------------------------------------------------------------------
# Spacing constants (pixels)
# ---------------------------------------------------------------------------
SPACING_XS:  int = 4
SPACING_SM:  int = 8
SPACING_MD:  int = 12
SPACING_LG:  int = 16
SPACING_XL:  int = 24
SPACING_XXL: int = 32

# ---------------------------------------------------------------------------
# Dimension constants (pixels)
# ---------------------------------------------------------------------------
BTN_HEIGHT_PRIMARY: int = 44   # Generate, Translate, Save, Back, Clear
BTN_HEIGHT_CANCEL:  int = 36   # Cancel button
BTN_HEIGHT_MINI:    int = 28   # colour-swatch buttons
BTN_WIDTH_BROWSE:   int = 80   # Browse… / Save as…
BTN_WIDTH_NARROW:   int = 100  # Open Folder and other small buttons
BTN_WIDTH_SWATCH:   int = 36   # colour-swatch square buttons
PROGRESS_BAR_HEIGHT: int = 16


# ---------------------------------------------------------------------------
# Widget-style helpers
# ---------------------------------------------------------------------------
# Each helper calls widget.configure() so colours are resolved at call time
# (not at import time), keeping them correct after appearance-mode changes.


def apply_entry_style(widget: object) -> None:
    """Apply input-field colours to *widget*."""
    widget.configure(  # type: ignore[union-attr]
        fg_color=p("input_bg"),
        text_color=p("text_primary"),
    )


def apply_accent_btn_style(widget: object) -> None:
    """Apply primary-accent colours to *widget* (Generate, Translate, Save)."""
    widget.configure(  # type: ignore[union-attr]
        fg_color=p("accent"),
        hover_color=p("accent_hov"),
    )


def apply_secondary_btn_style(widget: object) -> None:
    """Apply secondary colours to *widget* (Clear, Back, Browse, Open Folder)."""
    widget.configure(  # type: ignore[union-attr]
        fg_color=p("secondary"),
        hover_color=p("secondary_hov"),
    )


def apply_cancel_btn_style(widget: object) -> None:
    """Apply cancel/destructive colours to *widget*."""
    widget.configure(  # type: ignore[union-attr]
        fg_color=p("progress_err"),
        hover_color=p("secondary_hov"),
        text_color=("#FFFFFF", "#FFFFFF"),
    )


def apply_progress_bar_style(widget: object) -> None:
    """Apply idle progress-bar colour to *widget*."""
    widget.configure(  # type: ignore[union-attr]
        progress_color=p("progress_idle"),
    )


def apply_stage_label_style(widget: object) -> None:
    """Apply secondary-text colour to a stage/status *label*."""
    widget.configure(  # type: ignore[union-attr]
        text_color=p("text_secondary"),
    )


def apply_secondary_label_style(widget: object) -> None:
    """Apply secondary-text colour to informational *labels*."""
    widget.configure(  # type: ignore[union-attr]
        text_color=p("text_secondary"),
    )


def apply_settings_header_style(widget: object) -> None:
    """Apply subheader font and primary-text colour to a settings header *label*."""
    widget.configure(  # type: ignore[union-attr]
        font=font("subheader"),
        text_color=p("text_primary"),
    )


def apply_window_bg(window: object) -> None:
    """Apply background colour to the top-level *window*."""
    window.configure(  # type: ignore[union-attr]
        fg_color=p("bg"),
    )
```

Important notes:
- The `# type: ignore` comments are needed because the helpers accept a generic `object` to avoid importing CTk widget types (which are only available at runtime with a display).
- `apply_secondary_label_style` and `apply_stage_label_style` are equivalent in behaviour; both are exposed so call sites can use the most descriptive name for the widget being styled. Do NOT merge them into one — they serve different semantic roles and may diverge in the future.
- Do NOT import anything from customtkinter at module level (to preserve headless importability). Only `p` and `font` from `.theme` are needed.
